- Respuesta estudiante -

SELECT DISTINCT academico.nombre 
FROM tesis INNER JOIN academico ON academico.id = ref_tutor 
INNER JOIN estudiante ON estudiante.id = ref_estudiante INNER JOIN revisor ON revisor.ref_tesis = academico.id
WHERE anio_ingreso = 2019
ORDER BY nombre
